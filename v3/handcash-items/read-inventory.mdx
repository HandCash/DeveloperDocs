```typescript
import { HandCashConnect } from '@handcash/handcash-connect';

const connect = new HandCashConnect({ appId, appSecret });
const account = connect.getAccountFromAuthToken('<AUTH_TOKEN>');
const items = await account.items.getItemsInventory({ searchString: '', limit: 50 });
console.log(items);
```
