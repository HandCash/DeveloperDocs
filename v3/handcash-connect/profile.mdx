## Get current user profile

```typescript
import { getInstance, Connect } from '@handcash/sdk';

const sdk = getInstance({ appId: 'YOUR_APP_ID', appSecret: 'YOUR_APP_SECRET' });
const client = sdk.getAccountClient('<AUTH_TOKEN>');

const { data, error } = await Connect.getCurrentUserProfile({ client });
if (error) throw error;

console.log('Public:', data?.publicProfile);
console.log('Private:', data?.privateProfile);
```

## Get public profiles by alias/handle

```typescript
const { data } = await Connect.getPublicUserProfiles({
  client: sdk.client,
  query: { aliases: ['nosetwo', 'handcash'] },
});

for (const p of data?.items ?? []) {
  console.log(p.handle, p.displayName);
}
```
